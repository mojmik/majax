		SELECT post_title,post_name,post_content,pm1.`_thumbnail_id`  FROM
		(SELECT post_title,post_content,post_name 
			,MAX(CASE WHEN pm1.meta_key = '_thumbnail_id' then pm1.meta_value ELSE NULL END) as `_thumbnail_id`
			FROM wp_posts LEFT JOIN wp_postmeta pm1 ON ( pm1.post_id = ID) 
			WHERE post_id=id 
			AND post_status like 'publish' 
			AND post_type like ''			
			GROUP BY ID, post_title
			) AS pm1
			 WHERE post_name like 'jiny auto'
